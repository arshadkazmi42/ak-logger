## How to use?

```js
var log = require('ak-logger');

const person = { name: 'Gustavo', age: 23 }

// Printing debug log as an object
log.debug(person, "DEBUG LOGS");
/* Result

=====DEBUG LOGS=====
{ name: 'Gustavo', age: 23 }
=====DEBUG LOGS=====
*/

// Printing debug log as an string (stringify = true)
log.debug(person, "DEBUG_LOGS", true)
/* Result

=====DEBUG_LOGS=====
{"name":"Gustavo","age":23}
=====DEBUG_LOGS=====
*/


```
